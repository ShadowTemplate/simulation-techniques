80 #
81 # @param test_function the statistical function that will be used to test the data
82 # @param num_test number of test that will be done using the specified function
83 # @param distr1 distribution that will be checked
84 # @param distr2 reference distribution
85 # @param samples_param if the origin distribution is exponentially distributed this value represents

88 # @param alpha confidence level for the statistical test

90 # if the test_function is the Kolmogorov-Smirnov test it represents the parameters of the

92 def _run_test(test_function, num_test, distr1, distr2, samples_param, alpha, *args):
93  num_rej = 0
94  if test_function.__name__ == _chi_square_test.__name__:
95  for _ in range(num_test):
96  if not _chi_square_test(distr1.get_sample_data(samples_param), distr2.get_sample_data(samples_param), alpha, None if not args else args[0]):
97  num_rej += 1

99  for _ in range(num_test):

171 # @param dst_distr_name name of the reference distribution ('uniform' or 'norm')
172 # @param alpha significance level of the test
173 # @param params parameters for the theoretical distribution used
174 #
175 # @return True if the null hypothesis is not rejected, False otherwise
176 def _kolmogorov_smirnov_test(samples, dst_distr_name, alpha, params=None):
177  num_samples = len(samples)
178 
179  if dst_distr_name == 'uniform':
180  curr_args = (params[0], params[1]) if params else ()
181  elif dst_distr_name == 'norm':
182  curr_args = (params[0], params[1]) if params else ()
183  else:
184  raise ValueError('Unsupported destination distribution')
185 
186  if curr_args:
187  d, p_val = stats.kstest(samples, dst_distr_name, args=curr_args, mode="asymp")
188  else:
189  d, p_val = stats.kstest(samples, dst_distr_name, mode="asymp")
190 
191  # Computes the critical value considering the number of examples
192  if num_samples > 35:
193  critical_d = Constants.KS_CRITICAL_VALUES['over'][alpha] / sqrt(num_samples)
194  else:
195  critical_d = Constants.KS_CRITICAL_VALUES[num_samples][alpha]
196 
197  return d <= critical_d

200 ##
201 # Runs the chi-square test between the sample data specified and some
202 # other generated using a specific distribution using a specific value for the
203 # significance of the test.
204 #
205 # @param ref_samples reference values for the test
206 # @param obs_samples observed data for the test
207 # @param alpha significance level for the test
208 # @param num_int number of interval in which the values will be grouped
209 #
210 # @return True if the null hypothesis is not reject, False otherwise
211 def _chi_square_test(ref_samples, obs_samples, alpha, num_int=None):
212  samples_num = len(ref_samples)
213  test_value_present = False if Constants.CS_TEST_VALUES.get(samples_num - 1) is None else True
214 
215  # checks if the number of samples is the same and if
216  # is present a critical value for the specified significance level
217  assert samples_num == len(obs_samples) and test_value_present
218 
219  num_interval = num_int if num_int else math.ceil(sqrt(samples_num))
220 
221  # distributes appropriately the reference values in the chosen number of intervals
222  ref_samples = sorted(ref_samples)
223 
224  freq_values_ref = [math.floor(samples_num / num_interval) for _ in range(num_interval)]
225  if samples_num % num_interval != 0:
226  freq_values_ref[-1] += samples_num - num_interval * freq_values_ref[0]
227 
228  ranges_ref = [ref_samples[(i + 1) * freq_values_ref[0] - 1] if i != num_interval else ref_samples[-1] for i in range(num_interval)]
229 
230  freq_values_obs = [0 for _ in range(num_interval)]
231  obs_samples = sorted(obs_samples)
232 
233  curr_index = 0
234 
235  for data in obs_samples:
236  while data > ranges_ref[curr_index] and curr_index != num_interval - 1:
237  curr_index += 1
238  freq_values_obs[curr_index] += 1
239 
240  tot_sum = sum([pow(freq_values_obs[i] - freq_values_ref[i], 2) / freq_values_ref[i] for i in range(num_interval)])
241 
242  return tot_sum < Constants.CS_TEST_VALUES[samples_num - 1][alpha]

246 # Generates uniform data from an exponential distribution.
247 # If the arrival time {T1, T2, T3, ...} distribution is exponential then
248 # the arrival times are uniformly distributed on the interval (0,T).
249 # So the interarrival times are normalized using cumulative sums.
250 #
251 # @param distr specific exponential distribution
252 # @param max_time time interval in which will be generated the values
253 def _get_uniformed_exp_values(distr, max_time):
254  values = []
255  t = 0
256  while t < max_time:
257  values.append(distr.next())
258  t += values[-1]
259 
260  uniformed_values = []
261  curr_tot = 0
262 
263  for i in range(len(values) - 1):
264  curr_tot += values[i]
265  uniformed_values.append(curr_tot / max_time)
266 
267  return uniformed_values
